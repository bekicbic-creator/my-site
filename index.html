import pygame import random import sys import math pygame.init() # --- Настройки окна --- WIDTH, HEIGHT = 800, 600 win = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Burger Rush Pixel") # --- Цвета --- WHITE = (255, 255, 255) BLACK = (0, 0, 0) SKY = (150, 200, 255) GREEN = (80, 200, 80) YELLOW = (255, 230, 0) RED = (255, 80, 80) # --- Музыка и звуки --- boost_sound = pygame.mixer.Sound(pygame.mixer.Sound(file=None)) pygame.mixer.Sound.set_volume(boost_sound, 0.6) # --- Игровые параметры --- clock = pygame.time.Clock() gravity = 0.5 # --- Классы --- class Player: def __init__(self): self.x = WIDTH // 2 self.y = HEIGHT - 120 self.width = 32 self.height = 32 self.vel_y = 0 self.on_ground = False self.boosted = False self.boost_timer = 0 self.aura_radius = 0 def draw(self): color = (0, 150, 255) if not self.boosted else (255, 255, 0) pygame.draw.rect(win, color, (self.x, self.y, self.width, self.height)) if self.boosted: self.aura_radius = 50 + int(math.sin(pygame.time.get_ticks() * 0.01) * 5) pygame.draw.circle(win, (255, 255, 150), (self.x + 16, self.y + 16), self.aura_radius, 2) def move(self, keys): speed = 5 if not self.boosted else 9 if keys[pygame.K_a]: self.x -= speed if keys[pygame.K_d]: self.x += speed if self.x < 0: self.x = 0 if self.x > WIDTH - self.width: self.x = WIDTH - self.width def jump(self): if self.on_ground: self.vel_y = -10 self.on_ground = False def update(self): self.vel_y += gravity self.y += self.vel_y if self.y + self.height >= HEIGHT - 60: self.y = HEIGHT - 60 - self.height self.vel_y = 0 self.on_ground = True # таймер буста if self.boosted: self.boost_timer -= 1 if self.boost_timer <= 0: self.boosted = False class Burger: def __init__(self): self.x = random.randint(0, WIDTH - 32) self.y = -50 self.speed = random.uniform(2, 5) def update(self): self.y += self.speed if self.y > HEIGHT: self.y = -50 self.x = random.randint(0, WIDTH - 32) def draw(self): pygame.draw.rect(win, (210, 100, 30), (self.x, self.y, 32, 20)) pygame.draw.rect(win, (255, 220, 120), (self.x, self.y + 8, 32, 4)) pygame.draw.rect(win, GREEN, (self.x, self.y + 4, 32, 4)) # --- Функция столкновения --- def collide(a, b): return (a.x < b.x + 32 and a.x + a.width > b.x and a.y < b.y + 20 and a.y + a.height > b.y) # --- Основная функция игры --- def main(): player = Player() burgers = [Burger() for _ in range(4)] score = 0 font = pygame.font.SysFont("pixelmix", 28) run = True while run: clock.tick(60) win.fill(SKY) pygame.draw.rect(win, GREEN, (0, HEIGHT - 60, WIDTH, 60)) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: player.jump() keys = pygame.key.get_pressed() player.move(keys) player.update() # Обновление бургеров for burger in burgers: burger.update() burger.draw() if collide(player, burger): score += 1 burger.y = -50 burger.x = random.randint(0, WIDTH - 32) # буст player.boosted = True player.boost_timer = 60 * 15 // 10 # ~15 сек boost_sound.play() player.draw() # интерфейс score_text = font.render(f"Счёт: {score}", True, BLACK) win.blit(score_text, (10, 10)) if player.boosted: timer_text = font.render(f"Буст: {player.boost_timer//60}s", True, YELLOW) win.blit(timer_text, (10, 40)) pygame.display.update() main()
